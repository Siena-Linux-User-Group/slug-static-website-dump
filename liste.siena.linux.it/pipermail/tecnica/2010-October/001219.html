>
> lo questo script:
>
> ----------------------------------------
> #!/bin/bash
>
> CONTINUA=1 # varaiabile di controllo del ciclo while
>
> while [ $CONTINUA -eq 1 ]; do
>     # controllo
>     ls continua > /dev/null 2>&1
>     if [ ! $? -eq 0 ]; then
>         CONTINUA=0
>     fi
>
>     # do stuff here
>     date +%s
>     sleep 1
> done
>
> exit 0
> ----------------------------------------
>
> controlla che esista il file './continua': se esiste allora esegue 'do
> stuff here', altrimenti termina ;)

Grazie a tutti,
ho trovato la soluzione "casereccia":

...
	unsigned char buf[2], sign, strTemp[16];
	float temp;
	// lcdSetPos(ROW3);
	DS1631_readTemp(buf);
	// Check sign bit
	if((buf[0] > 0x80)) {
		sign = '-';
		/* Negate all bits */
		buf[0] = ~buf[0];
		buf[1] = ~buf[1] + 1;
	} else {
		sign = '+';	
	}	
	temp = buf[0];
	temp += ((buf[1] >> 4) * 0.0625);
	ftoa((float)temp, strTemp, 2, 'f');
	sprintf(tmpBuf,"t:%c%s°C ",sign, strTemp);
...
